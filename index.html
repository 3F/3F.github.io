namespace net.r_eg
{
    public class Info: ISource, IContact
    {
        public IEnumerable<SourceUri> PublicSources
        {
            get
            {
                return new SourceUri[] { 
                    new SourceUri("bitbucket.org/3F", SType.Primary),
                    new SourceUri("github.com/3F", SType.Mirror),
                };
            }
        }

        public IEnumerable<Project> hostedProjects(PType type)
        {
            if(type != PType.OpenSource) {
                throw new AccessException(
                    "Some old and very old projects(utilities, plugins, etc.) on Java, C++, C#, PHP5, JavaScript and others..",
                    "Are already lost its meaning or there are some problems to publish this for public",

                    "However, perhaps, a some logic will be transferred to primary sandbox for reuse later",
                    "Like this (with a vt100 terminal emulation), etc. "
                );
            }

            foreach(Project project in hostedProjects("Sourceforge.net", "MS Visual Studio Gallery", "*")) {
                yield return project;
            }
        }

        public List<IContactLeaf> getContacts()
        {
            return new List<IContactLeaf>()
            { 
                new Email("entry.reg@gmail.com"), 
                new Skype("rbsystem")
            };
        }
    }
}
